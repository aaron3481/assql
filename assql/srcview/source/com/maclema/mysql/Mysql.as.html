        public static function escapeString(str:String):String
        {
            if ( str.indexOf("\\") != -1 )
            {
                str.replace("\\", "\\"+"\\");
            }
            
            if ( str.indexOf("\'") != -1 )
            {
                str = str.replace("\'", "\\"+"\'");
            }
            
            if ( str.indexOf("\"") != -1 )
            {
                str = str.replace("\'", "\\"+"\"");
            }
            
            return str;
        }
